11pub fn get_config_dir() -> anyhow::Result<PathBuf> {
12    let app_strategy = etcetera::choose_app_strategy(AppStrategyArgs {
13        app_name: "gofile-cli".into(),
14        author: "".into(),
15        top_level_domain: "".into(),
16    })?;
17
18    let config_dir = app_strategy.config_dir();
19
20    // Create config dir if it does not exist.
21    std::fs::create_dir_all(&config_dir).context("failed to create config dir")?;
22
23    Ok(config_dir)
24}

26#[derive(Debug, argh::FromArgs)]
27#[argh(description = "a cli to interact with gofile")]
28struct Options {
29    #[argh(subcommand)]
30    subcommand: Subcommand,
31}
32
33#[derive(Debug, argh::FromArgs)]
34#[argh(subcommand)]
35enum Subcommand {
36    Download(self::commands::download::Options),
37    Config(self::commands::config::Options),
38    Upload(self::commands::upload::Options),
39    Info(self::commands::info::Options),
40}
41
42async fn async_main(options: Options) -> anyhow::Result<()> {
43    let client = gofile::Client::new();
44    match options.subcommand {
45        Subcommand::Download(options) => self::commands::download::exec(client, options).await?,
46        Subcommand::Config(options) => self::commands::config::exec(client, options).await?,
47        Subcommand::Upload(options) => self::commands::upload::exec(client, options).await?,
48        Subcommand::Info(options) => self::commands::info::exec(client, options).await?,
49    }
50
51    Ok(())
52}
53
54fn main() -> anyhow::Result<()> {
55    let options: Options = argh::from_env();
56
57    let tokio_rt = tokio::runtime::Builder::new_multi_thread()
58        .enable_all()
59        .build()?;
60    tokio_rt.block_on(async_main(options))?;
61
62    Ok(())
63}
